MameBlock.js

Simple visual code editing library.

  • Simple: just only 100 lines Javascript code with some library.
  • Embedded : You can use your own visual code editor with Web page and apps.
  • OpenSource : released under the MIT License.

Download (Beta ver0.1)

4.Javascript

			
Load jquery and library.

			
<!-- Javascript -->
<script type="text/javascript" src="js/jquery-1.11.2.min.js"></script>
<script type="text/javascript" src="js/mameblock.min.js"></script>
			

			
			
Load control script.

			
<script type="text/javascript">
$(function() {
    <!-- # Init -->
    var codeArea = $("#exampleArea");
    var codeText = MAME_BLOCK.init("#loading_area", "mameBlock_template.html #mame_template");
    /* wait for DOM building */
    var wait0 = setTimeout(function() {
        setCode(MAME_BLOCK.getCodeBlock(), codeArea);
    }, 500);
    
    <!-- # Edit -->
    $("#modal-window").mameModal({
        closeOnEscape: false
    });

    $('#mameEdit').click(function(e){
        MAME_BLOCK.copyCodeBlock("#mame_template");
        $('#modal-window')
            .trigger('openModal')
            .css("top", $(window).scrollTop() + 10);
        e.preventDefault();
    });
    
    $('#editor_close').click(function(e){
        $('#modal-window').trigger('closeModal');
        MAME_BLOCK.reloadCodeBlock('#loading_area');
        setCode(MAME_BLOCK.getCodeBlock(), codeArea);
        e.preventDefault();
    });

});
</script>
